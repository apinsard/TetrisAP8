38 void CProjetTetris::OnRender() {
39  //-------------------------------------------------------
40  // gestion du temps
41  //-------------------------------------------------------
42  float dMax = 2.0f; //Max difficulty
43  float dMin = 30.0f; //Min difficulty
44  float sMax = 2688.0f; //Max difficulty for score
45 
46  float d = max(dMax, dMax + (((dMin - dMax) / sMax) * (sMax - m_game.GetScore())));
47 
48  if (m_step > round(d))
49  m_step = 0;
50 
51  m_level = round((dMin - d)/2.0f);
52 
53  if(AR_GameOver == this->m_game.Update(m_step)) {
54  DrawGameOver();
55  return;
56  }
57 
58 
59  //-------------------------------------------------------
60  // commande utilisateur
61  //-------------------------------------------------------
62  if (m_UserEvents->m_keyBoardEvents[Key_a]) {
63  this->m_game.MovePiece(PA_RotateLeft);
64  }
65  if (m_UserEvents->m_keyBoardEvents[Key_z]) {
66  this->m_game.MovePiece(PA_RotateRight);
67  }
68  if (m_UserEvents->m_keyBoardEvents[Key_DOWN]) {
69  this->m_game.MovePiece(PA_MoveBottom);
70  }
71  if (m_UserEvents->m_keyBoardEvents[Key_RIGHT]) {
72  this->m_game.MovePiece(PA_MoveRight);
73  }
74  if (m_UserEvents->m_keyBoardEvents[Key_LEFT]) {
75  this->m_game.MovePiece(PA_MoveLeft);
76  }
77  if (m_UserEvents->m_keyBoardEvents[Key_SPACE]) {
78  this->m_game.MovePiece(PA_MoveBottom2);
79  }
80 
81  //-------------------------------------------------------
82  // affichage
83  //-------------------------------------------------------
84  // on passe en rendu 2d
85  m_renderer.Start2DRender();
86 
87  // affichage du score
88  DrawInfo();
89 
90  // affichage du tetris
91  DrawTetris();
92 
93  // fin du rendu 2d
94  m_renderer.End2DRender();
95 
96  // maj du compteur de passage dans cette fonction
97  m_step ++;
98 }

139 void CProjetTetris::DrawTetris() {
140 
141  vector<TGameRow> gameTable = m_game.GetBoard().GetGameTable();
142  float tetrisWidth = gameTable[0].size()* m_game.GetCaseDim();
143  float tetrisHeight = gameTable.size() * m_game.GetCaseDim();
144  float dim = m_game.GetCaseDim();
145 
146  // fond du tetris
147  DrawFillRect ( m_game.GetXPos(), m_game.GetYPos(), tetrisWidth, tetrisHeight, CVector3(0.0f, 0.0f, 0.0f));
148  //DrawFillRect ( m_game.GetXPos(), m_game.GetYPos(), tetrisWidth, tetrisHeight, CVector3(17.0f/255.0f,218.0f/255.0f,84.0f/255.0f) );
149 
150  // affichage du "cadre" tetris
151  DrawRect ( m_game.GetXPos()-0.1f, m_game.GetYPos()-0.1f, (tetrisWidth+0.2f), (tetrisHeight+0.2f), CVector3(213.0f/255.0f,213.0f/255.0f,213.0f/255.0f) );
152 
153  // affichage du tableau tetris
154  for (unsigned int i=0; i<gameTable.size(); i++){
155  for (unsigned int j=0; j<gameTable[i].size(); j++)
156  if(gameTable[i][j].m_used)
157  DrawFillSquare(m_game.GetXPos()+(j*dim), m_game.GetYPos()+(i*dim), dim, gameTable[i][j].m_color);
158  }
159 
160  // affichage de la pÓece en cours
161  CPieceAbstract* piece = this->m_game.GetPiece();
162  for (unsigned int i=0; i<piece->GetDim(); i++) {
163  for (unsigned int j=0; j<piece->GetDim(); j++) {
164  if (piece->GetTable()[i][j] == 1) {
165  DrawFillSquare(
166  this->m_game.GetXPos()+(piece->GetColIndex()+j)*dim,
167  this->m_game.GetYPos()+(piece->GetRowIndex()+i)*dim,
168  dim, piece->GetColor());
169  }
170  }
171  }
172 
173 }
