228 void CRenderer::DrawGround(float _size)
229 {
230  float step = _size/40;
231 
232  glDisable(GL_LIGHTING);
233  glColor3d(0.3,0.3,0.3);
234  unsigned int count = 0;
235  for (float i = 0 ; i <= 2.0*_size ; i=i+step, count++)
236  {
237  if (!(count%5)) glLineWidth(3.0f);
238  else glLineWidth(1.0f);
239  glBegin(GL_LINES);
240  glVertex3d(-_size+i,0.001,-_size);
241  glVertex3d(-_size+i,0.001,_size);
242  glVertex3d(-_size,0.001,-_size+i);
243  glVertex3d(_size,0.001,-_size+i);
244  glEnd();
245  }
246 
247  // draw some plane
248  glColor3d(0.4,0.4,0.4);
249  glBegin(GL_QUADS);
250  glVertex3d(-_size,-0.01,-_size);glVertex3d(-_size,-0.01,_size);
251  glVertex3d(_size,-0.01,_size);glVertex3d(_size,-0.01,-_size);
252  glEnd();
