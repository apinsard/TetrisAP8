29 void CGuiGlut::InitializeGraphics (int argc, char **argv, CApplication* application)
30 {
31  //
32  m_application = application;
33 
34  // initialize GLUT state based on command line arguments
35  glutInit (&argc, argv);
36 
37  // display modes: RGB+Z and double buffered
38  GLint mode = GLUT_RGB | GLUT_DEPTH | GLUT_DOUBLE;
39  glutInitDisplayMode (mode);
40 
41  // create and initialize our window with GLUT tools
42  // (center window on screen with size equal to "ws" times screen size)
43  const int sw = m_application->GetWindowInfo()->m_width; //glutGet (GLUT_SCREEN_WIDTH);
44  const int sh = m_application->GetWindowInfo()->m_height; //glutGet (GLUT_SCREEN_HEIGHT);
45  const float ws = 0.8f; // window_size / screen_size
46  const int ww = (int) (sw * ws);
47  const int wh = (int) (sh * ws);
48  glutInitWindowPosition ((int) (sw * (1-ws)/2), (int) (sh * (1-ws)/2));
49  glutInitWindowSize (ww, wh);
50  windowID = glutCreateWindow (appVersionName.c_str());
51  ReshapeFunc (ww, wh);
52  InitGL ();
53 
54  // register our display function, make it the idle handler too
55  glutDisplayFunc (&DisplayFunc);
56  glutIdleFunc (&DisplayFunc);
57 
58  // register handler for window reshaping
59  glutReshapeFunc (&ReshapeFunc);
60 
61  // register handler for keyboard events
62  glutKeyboardFunc (&KeyboardFunc);
63  glutSpecialFunc (&SpecialFunc);
64

67 
69  //glutMotionFunc (mouseMotionFunc);
70 
72  //glutPassiveMotionFunc (mousePassiveMotionFunc);
73 
75  //glutEntryFunc (mouseEnterExitWindowFunc);
76 
77  m_application->Init();
78 }

80 void CGuiGlut::InitGL(void)
81 {
82  // background = dark gray
83  glClearColor (0.3f, 0.3f, 0.3f, 0);

85  // OpenGL pour traiter les objets dans la scene a besoin de 2/3 routines, savoir dans quel ordre il affiche les éléments
86  // affiche t'on les faces caché (normale dans le sens de la caméra)
87  // http://www.linuxgraphic.org/section3d/openGL/didacticiels/didac2/didac4.html
88  glEnable(GL_DEPTH_TEST);

93  glShadeModel(GL_SMOOTH);
94 
95  // enable depth buffer clears
96  glClearDepth (1.0f);

99  //glShadeModel (GL_SMOOTH);

102  //glDepthFunc (GL_LESS);
103  //glEnable (GL_DEPTH_TEST);
104

107  //glCullFace (GL_BACK);
108 
109  // enable blending and set typical "blend into frame buffer" mode

119 void CGuiGlut::ReshapeFunc(int width, int height)
120 {
121  // set viewport to full window
122  glViewport(0, 0, width, height);
123 
124  // set perspective transformation
125  glMatrixMode (GL_PROJECTION);
126  glLoadIdentity ();
127  const GLfloat w = width;
128  const GLfloat h = height;
129  const GLfloat aspectRatio = (height == 0) ? 1 : w/h;
130  const GLfloat fieldOfViewY = 45;
131  const GLfloat hither = 1; // put this on CCamera so PlugIns can frob it
132  const GLfloat yon = 400; // put this on CCamera so PlugIns can frob it
133  gluPerspective (fieldOfViewY, aspectRatio, hither, yon);
134 
135  // leave in modelview mode
136  glMatrixMode(GL_MODELVIEW);
137 }
