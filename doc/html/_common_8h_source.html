126  static std::vector<std::string> Split(const std::string & src, const char delim)
127  {
128  std::vector<std::string> v;
129  std::string::const_iterator p = src.begin ();
130  std::string::const_iterator q =
131  find (p, src.end (), delim);
132 
133  while (q != src.end ())
134  {
135  v.push_back (std::string (p, q));
136  p = q;
137  q = find (++p, src.end (), delim);
138  }
139  if (p != src.end ())
140  v.push_back (std::string (p, src.end ()));
141 
142  return v;
143  }
