176 float CMatrix44::det()
177 {
178  float d = (M00 * M11 - M01 * M10) * (M22 * M33 - M23 * M32) -
179  (M00 * M12 - M02 * M10) * (M21 * M33 - M23 * M31) +
180  (M00 * M13 - M03 * M10) * (M21 * M32 - M22 * M31) +
181  (M01 * M12 - M02 * M11) * (M20 * M33 - M23 * M30) -
182  (M01 * M13 - M03 * M11) * (M20 * M32 - M22 * M30) +
183  (M02 * M13 - M03 * M12) * (M20 * M31 - M21 * M30);
184 
185  return d;
186 }

320 {

332  m[0][1] = static_cast<float>(u.x); m[1][1] = static_cast<float>(u.y) ; m[2][1] = static_cast<float>(u.z); m[3][1] = 0.0f;
333  m[0][2] = static_cast<float>(-f.x); m[1][2] = static_cast<float>(-f.y); m[2][2] = static_cast<float>(-f.z); m[3][2] = 0.0f;
334  m[0][3] = 0.0f; m[1][3] = 0.0f; m[2][3] = 0.0f; m[3][3] = 1.0f;

351 {
352  float cx (right-left) ;

354  float cz (farVal-nearVal) ;
